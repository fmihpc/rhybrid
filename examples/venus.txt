# Python examples using Analysator

export PYTHONPATH=$PYTHONPATH:~/bin/analysator/

# create videos of 2d pseudocolor slices

# individual frames
mkdir -p png/
python plotter_rhybrid_2d_slice.py 2 ../ "run1" 6051.8e3 0 20000
# put together a video
ffmpeg -framerate 5 -pattern_type glob -i './png/Hsw_n_*.png' -c:v libx264 -crf 18 -pix_fmt yuv420p -vf scale=1920:-2 Hsw_n.mp4
ffmpeg -framerate 5 -pattern_type glob -i './png/B_*.png' -c:v libx264 -crf 18 -pix_fmt yuv420p -vf scale=1920:-2 B.mp4
ls Hsw_n.mp4 B.mp4
